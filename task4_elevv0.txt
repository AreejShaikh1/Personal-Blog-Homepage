      { title: "Trip to Mohenjo-Daro", category: "Travel", description: "A walk through the ruins of the ancient Indus Valley Civilization.", date: "2025-06-20", image: "https://picsum.photos/400/200?travel2" },
      { title: "Street Food in Pakistan", category: "Food", description: "Pakistani street food adventures you can‚Äôt miss.", date: "2025-06-10", image: "https://picsum.photos/400/200?food2" },
      { title: "Future of Programming", category: "Tech", description: "Which coding languages will dominate in the next decade?", date: "2025-05-30", image: "https://picsum.photos/400/200?tech2" }
    ];

    const postsPerPage = 3;
    let currentPage = 1;

    function displayPosts() {
      const container = document.getElementById("blogContainer");
      const category = document.getElementById("categoryFilter").value;
      const search = document.getElementById("searchInput").value.toLowerCase();

      let filteredPosts = posts.filter(p =>
        (category === "all" || p.category === category) &&
        p.title.toLowerCase().includes(search)
      );

      const start = (currentPage - 1) * postsPerPage;
      const end = start + postsPerPage;
      const paginatedPosts = filteredPosts.slice(start, end);

      container.innerHTML = paginatedPosts.map(p => `
        <div class="card">
          <img src="${p.image}" alt="${p.title}">
          <div class="card-content">
            <h3>${p.title}</h3>
            <p>${p.description}</p>
            <p class="date">${p.date}</p>
          </div>
        </div>
      `).join("");
    }

    function nextPage() {
      const totalPages = Math.ceil(posts.length / postsPerPage);
      if (currentPage < totalPages) {
        currentPage++;
        displayPosts();
      }
    }

    function prevPage() {
      if (currentPage > 1) {
        currentPage--;
        displayPosts();
      }
    }

    document.getElementById("categoryFilter").addEventListener("change", () => {
      currentPage = 1;
      displayPosts();
    });

    document.getElementById("searchInput").addEventListener("input", () => {
      currentPage = 1;
      displayPosts();
    });

    displayPosts();
  </script>
