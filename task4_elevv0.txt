  <script>
    const posts = [
      { title: "Tech in 2025", category: "Tech", image: "https://via.placeholder.com/400x200?text=Tech", description: "Exploring the future of technology.", date: "2025-08-01" },
      { title: "My Trip to Karachi", category: "Travel", image: "https://via.placeholder.com/400x200?text=Travel", description: "A wonderful journey to Karachi.", date: "2025-07-15" },
      { title: "Best Biryani in Sindh", category: "Food", image: "https://via.placeholder.com/400x200?text=Food", description: "Tasting the best biryani in Sukkur.", date: "2025-06-25" },
      { title: "AI & Future Jobs", category: "Tech", image: "https://via.placeholder.com/400x200?text=AI", description: "How AI will impact careers.", date: "2025-05-10" },
      { title: "Trip to Islamabad", category: "Travel", image: "https://via.placeholder.com/400x200?text=Islamabad", description: "Exploring Islamabad‚Äôs beauty.", date: "2025-04-20" },
      { title: "Street Food in Karachi", category: "Food", image: "https://via.placeholder.com/400x200?text=Foodie", description: "Street food adventures in Karachi.", date: "2025-03-12" },
    ];

    const postsPerPage = 3;
    let currentPage = 1;

    const postsGrid = document.getElementById("postsGrid");
    const categoryFilter = document.getElementById("categoryFilter");
    const searchInput = document.getElementById("searchInput");
    const prevPageBtn = document.getElementById("prevPage");
    const nextPageBtn = document.getElementById("nextPage");

    function renderPosts() {
      postsGrid.innerHTML = "";
      let filteredPosts = posts.filter(post =>
        (categoryFilter.value === "All" || post.category === categoryFilter.value) &&
        post.title.toLowerCase().includes(searchInput.value.toLowerCase())
      );

      const start = (currentPage - 1) * postsPerPage;
      const end = start + postsPerPage;
      const paginatedPosts = filteredPosts.slice(start, end);

      paginatedPosts.forEach(post => {
        const card = document.createElement("div");
        card.className = "card";
        card.innerHTML = `
          <img src="${post.image}" alt="${post.title}" />
          <div class="card-content">
            <h3>${post.title}</h3>
            <p>${post.description}</p>
            <small>${post.date} | ${post.category}</small>
          </div>
        `;
        postsGrid.appendChild(card);
      });

      prevPageBtn.disabled = currentPage === 1;
      nextPageBtn.disabled = end >= filteredPosts.length;
    }

    categoryFilter.addEventListener("change", () => {
      currentPage = 1;
      renderPosts();
    });

    searchInput.addEventListener("input", () => {
      currentPage = 1;
      renderPosts();
    });

    prevPageBtn.addEventListener("click", () => {
      if (currentPage > 1) {
        currentPage--;
        renderPosts();
      }
    });

    nextPageBtn.addEventListener("click", () => {
      currentPage++;
      renderPosts();
    });

    renderPosts();
  </script>
